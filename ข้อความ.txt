async function doSignin(){
  const u=document.getElementById('li-user').value.trim();

  if(!u||!p){showAlert('⚠️ กรุณากรอกข้อมูลให้ครบ');return;}
  // เช็ค local ก่อน (เร็วกว่า)
  const users=JSON.parse(localStorage.getItem('users')||'[]');
  let found=users.find(x=>x.username===u&&x.password===p);
  if(!found){
    // ถ้าไม่เจอใน local → เช็คจาก Google Sheet
    try {
      showAlert('🔄 กำลังตรวจสอบ...');
      const res = await gsGet({action:'login',username:u,password:p});
      if(res && res.status==='success' && res.data){
        found = res.data;
        // บันทึกลง local เพื่อใช้ครั้งหน้า
        const localUsers=JSON.parse(localStorage.getItem('users')||'[]');
        if(!localUsers.find(x=>x.username===u)){
          localUsers.push(found);
          localStorage.setItem('users',JSON.stringify(localUsers));
        }
      }
    } catch(e){}
  }
  if(!found){showAlert('❌ ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง');return;}

  showAlert('✅ ยินดีต้อนรับ '+currentUser.fullname+' 🌸');
}

async function submitOrder(){

  const subject=document.getElementById('or-subject').value.trim();
  const deadline=document.getElementById('or-deadline').value;
  const detail=document.getElementById('or-detail').value.trim();
  if(!name||!lineId){showAlert('⚠️ กรุณากรอกชื่อและไอดีไลน์');return;}
  if(selectedSvcs.length===0){showAlert('⚠️ กรุณาเลือกประเภทงาน');return;}
  if(!deadline||!detail){showAlert('⚠️ กรุณากรอกกำหนดส่งและรายละเอียด');return;}

  const typeStr=selectedSvcs.map(id=>services.find(s=>s.id===id)?.name||id).join(' + ');

  orders.push(newOrder);
  localStorage.setItem('shop-orders', JSON.stringify(orders));
  // บันทึกไป Google Sheet
  try {
    await gsCall({action:'addOrder', data:{
      id: newOrder.id,
      name: newOrder.name,
      username: newOrder.username||'',
      lineId: newOrder.lineId||'',
      type: newOrder.type,
      subject: newOrder.subject,
      deadline: newOrder.deadline,
      pay: newOrder.pay,
      status: newOrder.status,
    }});
  } catch(e){}
  // เพิ่ม XP ให้ลูกค้า
  if(currentUser){
    addUserXP(currentUser.username, levelConfig.xpPerOrder, 'สั่งงาน: '+typeStr).then(()=>{
      // refresh level page ถ้ากำลังอยู่หน้า level
      if(document.getElementById('page-level').classList.contains('active')) renderLevelPage();
    });
  }

// ═══════════════ TELEGRAM ═══════════════
function sendTelegram(name,lineId,type,subject,deadline,detail,pay,price,discountInfo){
  // ใช้ hardcoded token ทุกเครื่องส่งได้เหมือนกัน
  const token = TG_TOKEN;
  const chatId = TG_CHATID;
  if(!token||!chatId) return;
  const discLine = discountInfo ? `\n🎁 ส่วนลด: ${discountInfo}` : '';
  const priceLine = price ? `\n💰 ราคาประเมิน: ${price}` : '';
  const msg=`🌸 *คำสั่งซื้อใหม่!*\n👤 ชื่อ: ${name}\n💬 Line: ${lineId||'-'}\n📌 งาน: ${type}\n📚 วิชา: ${subject||'-'}\n📅 กำหนดส่ง: ${deadline}\n💳 ชำระ: ${pay==='cash'?'เงินสด':'โอนจ่าย'}${priceLine}${discLine}\n📝 รายละเอียด: ${detail}`;
  fetch(`https://api.telegram.org/bot${token}/sendMessage`,{
    method:'POST',headers:{'Content-Type':'application/json'},
    body:JSON.stringify({chat_id:chatId,text:msg,parse_mode:'Markdown'})
  }).catch(()=>{});
}

// ═══════════════ PRICE CALCULATOR ═══════════════
function updatePriceCalc(){
  const box = document.getElementById('price-calc-box');
  const inputsDiv = document.getElementById('calc-inputs');
  if(!box || !inputsDiv) return;
  if(selectedSvcs.length === 0){ box.style.display='none'; return; }
  box.style.display='block';

  // Build calc inputs for selected services
  let html = '';
  selectedSvcs.forEach(id=>{
    const svc = services.find(s=>s.id===id);
    if(!svc) return;
    if(svc.calcType === 'perpage'){
      html += `<div style="margin-bottom:10px">
        <div style="font-size:.8rem;font-weight:600;color:#555555;margin-bottom:5px">${svc.icon} ${svc.name}</div>
        <div style="display:flex;gap:8px;align-items:center;flex-wrap:wrap">
          <input id="pages-${id}" type="number" min="1" placeholder="จำนวนหน้า" 
            oninput="calcTotal()"
            style="width:110px;padding:8px 10px;border:1.5px solid #ffb3d1;border-radius:8px;font-size:.84rem;background:#fff0f6;outline:none"/>
          <select id="density-${id}" onchange="calcTotal()" style="padding:8px 10px;border:1.5px solid #ffb3d1;border-radius:8px;font-size:.82rem;background:#fff0f6">
            <option value="${svc.pricePerPage||7}">น้อย (${svc.pricePerPage||7}฿/หน้า)</option>
            <option value="${svc.pricePerPageHigh||12}">เยอะ (${svc.pricePerPageHigh||12}฿/หน้า)</option>
          </select>
        </div>
      </div>`;
    } else {
      // fixed price
      const displayPrice = svc.basePrice || svc.pricePerPage || 0;
      html += `<div style="margin-bottom:6px;font-size:.82rem;color:#555555">${svc.icon} ${svc.name}: <strong style="color:#ff4d8d">${displayPrice} ฿</strong></div>`;
    }
  });
  inputsDiv.innerHTML = html;
  calcTotal();
}

function calcTotal(){
  let base = 0;
  let hasPerpage = false;
  let perpageTotal = 0;
  let perpagePages = 0;
  let perpageRate = 0;

  selectedSvcs.forEach(id=>{
    const svc = services.find(s=>s.id===id);
    if(!svc) return;
    if(svc.calcType==='perpage'){
      const pages = parseInt(document.getElementById('pages-'+id)?.value||0);
      const rate  = parseInt(document.getElementById('density-'+id)?.value||svc.pricePerPage);
      perpageTotal += pages * rate;
      perpagePages += pages;
      perpageRate = rate;
      hasPerpage = true;
    } else {
      base += svc.basePrice||0;
    }
  });

  // คำนวณ threshold: จำนวนหน้าที่ยังติดราคาขั้นต่ำ
  // เช่น min=20, rate=7 → ceil(20/7)=3 → 1-3 หน้าติดขั้นต่ำ, 4+ไม่ติด
  let minApplied = false;
  let minNote = '';
  if(hasPerpage && perpagePages > 0){
    const threshold = perpageRate > 0 ? Math.ceil(minOrderPrice / perpageRate) : 0;
    if(perpagePages <= threshold && perpageTotal < minOrderPrice){
      // ติดราคาขั้นต่ำ
      base += minOrderPrice;
      minApplied = true;
      minNote = 'ราคาขั้นต่ำ ' + minOrderPrice + ' ฿ (สำหรับ ' + threshold + ' หน้าแรก)';
    } else {
      // 4 หน้าขึ้นไป คิดราคาปกติ
      base += perpageTotal;
    }
  }

  calcBasePrice = base;
  if(base === 0){ document.getElementById('calc-result').style.display='none'; return; }
  document.getElementById('calc-result').style.display='block';
  document.getElementById('calc-before').textContent = base + ' ฿';

  // แสดง/ซ่อน note ขั้นต่ำ
  const minNoteEl = document.getElementById('calc-min-note');
  if(minNoteEl){
    if(minApplied){
      minNoteEl.style.display='block';
      minNoteEl.textContent = '⚠️ ' + minNote;
    } else {
      minNoteEl.style.display='none';
    }
  }

  // apply discount if any
  let total = base;
  const discRow = document.getElementById('calc-discount-row');
  if(appliedDiscount && base >= (appliedDiscount.minOrder||0)){
    let discAmt = appliedDiscount.type==='percent' ? Math.round(base*appliedDiscount.value/100) : appliedDiscount.value;
    total = Math.max(0, base - discAmt);
    document.getElementById('calc-discount-label').textContent = `ส่วนลด (${appliedDiscount.code})`;
    document.getElementById('calc-discount-amt').textContent = '-' + discAmt + ' ฿';
    discRow.style.display = 'flex';
  } else {
    discRow.style.display = 'none';
  }
  // enforce minimum order price

function applyCoupon(){
  const code = (document.getElementById('or-coupon').value||'').trim().toUpperCase();
  const res = document.getElementById('coupon-result');
  if(!code){ res.style.display='none'; return; }
  const found = discountCodes.find(d=>d.code===code);
  if(!found){
    appliedDiscount=null;
    res.style.display='block';

    calcTotal(); return;
  }
  // เช็ค allowInMin - โค้ดนี้ใช้ในราคาขั้นต่ำได้ไหม
  const minNoteEl = document.getElementById('calc-min-note');
  const isMinApplied = minNoteEl && minNoteEl.style.display !== 'none' && minNoteEl.textContent.includes('ขั้นต่ำ');
  if(isMinApplied && found.allowInMin === false){
    res.style.display='block';
    res.innerHTML='<span style="color:#e67e22">⚠️ โค้ดนี้ไม่สามารถใช้กับราคาขั้นต่ำได้</span>';
    appliedDiscount=null; calcTotal(); return;
  }
  if(calcBasePrice>0 && calcBasePrice < (found.minOrder||0)){
    res.style.display='block';
    res.innerHTML='<span style="color:#e74c3c">❌ ต้องสั่งงานขั้นต่ำ '+found.minOrder+' ฿ จึงจะใช้โค้ดนี้ได้</span>';
    appliedDiscount=null; calcTotal(); return;
  }
  appliedDiscount = found;
  res.style.display='block';
  const desc = found.type==='percent' ? 'ลด '+found.value+'%' : 'ลด '+found.value+' ฿';

}

// ─── Google Sheet API ───
async function gsCall(payload) {
  try {
    await fetch(GS_URL, {
      method: 'POST',
      headers: {'Content-Type':'text/plain'},
      body: JSON.stringify(payload),
      mode: 'no-cors'
    });
    return {ok: true};
  } catch(e) { return {ok: false}; }
}

async function gsGet(params) {
  try {
    const url = GS_URL + '?' + new URLSearchParams(params);
    const res = await fetch(url);
    if(!res.ok) return null;
    return await res.json();
  } catch(e) { return null; }
}

// ─── XP functions ───
async function getUserXP(username) {
  // ดึงจาก localStorage ก่อน (เร็วกว่า + ไม่ขึ้น network)
  const local = JSON.parse(localStorage.getItem('xp-' + username) || 'null');
  if (local) return local;
  // ถ้าไม่มีใน local ลอง Google Sheet
  try {
    const data = await gsGet({action:'getXP', username});
    if (data && typeof data.xp !== 'undefined') {
      localStorage.setItem('xp-' + username, JSON.stringify(data));
      return data;
    }
  } catch(e) {}
  return {xp:0, level:1, history:[]};
}

async function addUserXP(username, amount, reason) {
  try {
    const data = await getUserXP(username);
    const newXP = (data.xp || 0) + amount;
    const newLevel = calcLevelFromXP(newXP);
    const history = data.history || [];
    history.unshift({xp: amount, reason, date: new Date().toLocaleDateString('th-TH'), total: newXP});
    if(history.length > 20) history.pop();

    // บันทึก localStorage ก่อน (ทำงานทันที)
    const updated = {xp: newXP, level: newLevel, history};
    localStorage.setItem('xp-' + username, JSON.stringify(updated));

    // ส่งไป Google Sheet แบบ background (ไม่ blocking)
    gsCall({action:'addXP', username, xp: amount, reason}).catch(()=>{});

    // เช็ค level up
    const oldLevel = data.level || 1;
    if (newLevel > oldLevel) {
      const tier = levelConfig.tiers.find(t=>t.level===newLevel);
      if(tier) {
        showAlert('🎉 ยินดีด้วย! ขึ้นเป็น ' + tier.icon + ' ' + tier.name + ' แล้ว!');
        if (tier.reward) setTimeout(()=>showAlert('🎁 รางวัล: ' + tier.reward), 2000);
      }
    }
    return updated;
  } catch(e) {
    console.error('addUserXP error:', e);
    return {xp:0, level:1, history:[]};
  }
}

function calcLevelFromXP(xp) {
  let lv = 1;
  for (const t of levelConfig.tiers) {
    if (xp >= t.minXP) lv = t.level;
  }
  return lv;
}

function getTierByLevel(lv) {
  return levelConfig.tiers.find(t=>t.level===lv) || levelConfig.tiers[0];
}

// ═══════════════ SYNC DISCOUNT CODES FROM GOOGLE SHEET ═══════════════
async function syncDiscountCodesFromSheet(){
  try {
    const data = await gsGet({action:'getCodes'});
    if(!data || !data.data || !Array.isArray(data.data)) return;
    if(data.data.length === 0) return;
    // แปลงข้อมูลจาก Sheet เป็น format ที่ใช้
    const sheetCodes = data.data.map(row => ({
      code:       (row.code||'').toString().toUpperCase(),
      type:       row.type||'fixed',
      value:      parseFloat(row.value)||0,
      minOrder:   parseFloat(row.minOrder)||0,
      allowInMin: row.allowInMin === true || row.allowInMin === 'true' || row.allowInMin === 'TRUE',
    })).filter(d => d.code);
    // รวมกับ local - Sheet มีสิทธิ์สูงกว่า
    sheetCodes.forEach(sc => {
      if(!discountCodes.find(d=>d.code===sc.code)){
        discountCodes.push(sc);
      }
    });
    localStorage.setItem('discount-codes', JSON.stringify(discountCodes));
    renderDiscountCodes();
    console.log('✅ โหลดโค้ดส่วนลดจาก Sheet แล้ว:', sheetCodes.length, 'โค้ด');
  } catch(e) {
