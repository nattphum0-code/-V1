function closeNotice() {
  const mute = document.getElementById('notice-mute-chk').checked;
  if (mute) {
    localStorage.setItem('notice-mute-until', Date.now() + 24*60*60*1000);
  }
  document.getElementById('notice-overlay').classList.add('hidden');
}
function checkNotice() {
  const muteUntil = parseInt(localStorage.getItem('notice-mute-until') || '0');
  if (Date.now() < muteUntil) {
    document.getElementById('notice-overlay').classList.add('hidden');
  }
}

// ── LOGIN ──
function doLogin() {
  const user = document.getElementById('login-user').value.trim();
  const pass = document.getElementById('login-pass').value.trim();
  if (!user || !pass) { showAlert('⚠️ กรุณากรอกข้อมูลให้ครบ'); return; }
  const users = JSON.parse(localStorage.getItem('users') || '[]');
  const found = users.find(u => u.username === user && u.password === pass);
  if (found) {
    closeModal('login');
    showAlert('✅ ยินดีต้อนรับ ' + found.fullname + ' 🌸');
  } else {
    showAlert('❌ ชื่อผู้ใช้หรือรหัสผ่านไม่ถูกต้อง');
  }
}

// ── REGISTER ──
function doRegister() {
  const fullname = document.getElementById('reg-fullname').value.trim();
  const user = document.getElementById('reg-user').value.trim();
  const pass = document.getElementById('reg-pass').value.trim();
  const pass2 = document.getElementById('reg-pass2').value.trim();
  if (!fullname || !user || !pass || !pass2) { showAlert('⚠️ กรุณากรอกข้อมูลให้ครบ'); return; }
  if (pass.length < 6) { showAlert('⚠️ รหัสผ่านต้องมีอย่างน้อย 6 ตัว'); return; }
  if (pass !== pass2) { showAlert('❌ รหัสผ่านไม่ตรงกัน'); return; }
  const users = JSON.parse(localStorage.getItem('users') || '[]');
  if (users.find(u => u.username === user)) { showAlert('❌ ชื่อผู้ใช้นี้ถูกใช้ไปแล้ว'); return; }
  users.push({ fullname, username: user, password: pass });
  localStorage.setItem('users', JSON.stringify(users));
  closeModal('login');
  showAlert('🎉 สมัครสมาชิกสำเร็จ! ยินดีต้อนรับ ' + fullname + ' 🌸');
}

// ── ADMIN LOGIN ──
function doAdminLogin() {
  const user = document.getElementById('admin-user').value.trim();
  const pass = document.getElementById('admin-pass').value.trim();
  if (user === 'admin' && pass === 'admin1234') {
    closeModal('login');
    showAlert('✅ เข้าสู่ระบบแอดมินสำเร็จ!');
    const nav = document.querySelector('.nav-links');
    if (!document.getElementById('admin-nav-btn')) {
      const btn = document.createElement('button');
      btn.id = 'admin-nav-btn';

      btn.onclick = function() { showPage('admin', this); };
      nav.insertBefore(btn, nav.lastElementChild);
    }
    showPage('admin', document.getElementById('admin-nav-btn'));
  } else {
    showAlert('❌ รหัสผ่านไม่ถูกต้อง');
  }
}

// ── SERVICE CHECKBOX (max 2) ──
function toggleSvc(label, val) {
  const chk = label.querySelector('input');
  // prevent default checkbox then handle manually
  if (!label.classList.contains('selected')) {
    if (selectedServices.length >= 2) {
      chk.checked = false;
      showAlert('⚠️ เลือกได้สูงสุด 2 งานเท่านั้น');
      return;
    }
    label.classList.add('selected');
    chk.checked = true;
    selectedServices.push(val);
  } else {
    label.classList.remove('selected');
    chk.checked = false;
    selectedServices = selectedServices.filter(s => s !== val);
  }
}

// ── SELECT SERVICE FROM CARD ──
function selectService(type) {
  // Reset first
  selectedServices = [];
  document.querySelectorAll('.svc-check-label').forEach(l => {
    l.classList.remove('selected');
    l.querySelector('input').checked = false;
  });
  // Select the one clicked
  const target = document.querySelector(`.svc-check-label input[value="${type}"]`);
  if (target) {
    target.checked = true;
    target.closest('.svc-check-label').classList.add('selected');
    selectedServices = [type];
  }
  openModal('order');
}

// ── SUBMIT ORDER ──
function submitOrder() {
  const name = document.getElementById('order-name').value.trim();
  const subject = document.getElementById('order-subject').value.trim();
  const deadline = document.getElementById('order-deadline').value;
  const detail = document.getElementById('order-detail').value.trim();
  if (!name || selectedServices.length === 0 || !deadline || !detail) {
    showAlert('⚠️ กรุณากรอกข้อมูลที่จำเป็นให้ครบ'); return;
  }
  const typeStr = selectedServices.join(' + ');
  orders.push({id: orders.length + 1, name, type: typeStr, subject: subject || '-', deadline, status: 'pending'});
  // Telegram notification
  sendTelegramNotification(name, typeStr, subject, deadline, detail);
  // Reset
  document.getElementById('order-name').value = '';
  document.getElementById('order-subject').value = '';
  document.getElementById('order-deadline').value = '';
  document.getElementById('order-detail').value = '';
  document.getElementById('order-budget').value = '';
  selectedServices = [];
  document.querySelectorAll('.svc-check-label').forEach(l => { l.classList.remove('selected'); l.querySelector('input').checked = false; });
  closeModal('order');
  showAlert('🎉 ส่งคำสั่งซื้อสำเร็จ! เราจะติดต่อกลับเร็วๆ นี้ค่ะ');
}

// ── TELEGRAM ──
function sendTelegramNotification(name, type, subject, deadline, detail) {
  const token = localStorage.getItem('tg-token') || '';
  const chatId = localStorage.getItem('tg-chatid') || '';
  if (!token || !chatId) return;
  const msg = `🌸 *คำสั่งซื้อใหม่!*\n👤 ชื่อ: ${name}\n📌 งาน: ${type}\n📚 วิชา: ${subject||'-'}\n📅 กำหนดส่ง: ${deadline}\n📝 รายละเอียด: ${detail}`;
  fetch(`https://api.telegram.org/bot${token}/sendMessage`, {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({chat_id: chatId, text: msg, parse_mode: 'Markdown'})
  }).catch(() => {});
}

function saveTelegram() {
  const token = document.getElementById('tg-token').value.trim();
  const chatId = document.getElementById('tg-chatid').value.trim();
  if (!token || !chatId) { showAlert('⚠️ กรุณากรอก Token และ Chat ID'); return; }
  localStorage.setItem('tg-token', token);
  localStorage.setItem('tg-chatid', chatId);
  showAlert('✅ บันทึกการตั้งค่า Telegram แล้ว!');
}

function loadTelegramSettings() {
  const token = localStorage.getItem('tg-token') || '';
  const chatId = localStorage.getItem('tg-chatid') || '';
  document.getElementById('tg-token').value = token;
  document.getElementById('tg-chatid').value = chatId;
}

// ── PRICE SETTINGS ──
function buildPriceSettings() {
  const grid = document.getElementById('price-settings-grid');
  const entries = Object.entries(prices);
  grid.innerHTML = entries.map(([k, v]) => `
    <div class="price-item">
      <label>${k}</label>
      <input id="ps-${k}" value="${v}" placeholder="เช่น เริ่มต้น ฿99"/>
    </div>`).join('');
}

function savePrices() {
  Object.keys(prices).forEach(k => {
    const el = document.getElementById('ps-' + k);
    if (el) {
      prices[k] = el.value;
      const priceEl = document.getElementById('price-' + k);
      if (priceEl) priceEl.textContent = el.value;
    }
  });
  showAlert('✅ บันทึกราคาเรียบร้อย!');
}

// ── REVIEWS ──
function setStar(n) {
  selectedStar = n;
  document.querySelectorAll('.star-rating button').forEach((b, i) => b.classList.toggle('active', i < n));
}
setStar(5);

function submitReview() {
  const name = document.getElementById('review-name').value.trim() || 'ไม่ระบุชื่อ';
  const service = document.getElementById('review-service').value;
  const text = document.getElementById('review-text').value.trim();
  if (!text) { showAlert('⚠️ กรุณาเขียนรีวิวก่อนนะคะ'); return; }
  const now = new Date();
  reviews.unshift({name, service, stars: selectedStar, text, date: now.toLocaleDateString('th-TH',{day:'numeric',month:'short',year:'numeric'})});
  closeModal('review');
  renderReviews();
  showAlert('💖 ขอบคุณสำหรับรีวิวค่ะ!');
  document.getElementById('review-name').value = '';
  document.getElementById('review-text').value = '';
}

function renderReviews() {
  const makeTicker = r => `
    <div class="ticker-card">
      <div class="t-header"><div class="t-avatar">${r.name[0]}</div>
        <div><div class="t-name">${r.name}</div><div class="t-svc">${r.service}</div></div>
      </div>
      <div class="t-stars">${'★'.repeat(r.stars)}${'☆'.repeat(5-r.stars)}</div>
      <div class="t-text">${r.text}</div>
    </div>`;
  const ticker = document.getElementById('home-reviews-ticker');
  if (ticker) { const c = reviews.map(makeTicker).join(''); ticker.innerHTML = c + c; }
}

// ── ADMIN ──
function refreshAdmin() {
  const pending = orders.filter(o => o.status==='pending').length;
  const done = orders.filter(o => o.status==='done').length;
  document.getElementById('stat-orders').textContent = orders.length;
  document.getElementById('stat-pending').textContent = pending;
  document.getElementById('stat-done').textContent = done;
  const avg = reviews.length ? (reviews.reduce((s,r)=>s+r.stars,0)/reviews.length).toFixed(1) : '5.0';
  document.getElementById('stat-rating').textContent = avg;
  const sl = {pending:'รอดำเนินการ',doing:'กำลังทำ',done:'ส่งแล้ว'};
  const sc = {pending:'status-pending',doing:'status-doing',done:'status-done'};
  document.getElementById('admin-table-body').innerHTML = orders.map(o => `
    <tr>
      <td>#${o.id}</td><td><b>${o.name}</b></td><td>${o.type}</td><td>${o.subject}</td><td>${o.deadline}</td>
      <td><span class="status-badge ${sc[o.status]}">${sl[o.status]}</span></td>
      <td><select onchange="changeStatus(${o.id},this.value)" style="padding:5px 8px;border-radius:8px;border:1.5px solid var(--pink-200);background:var(--pink-50);font-size:.78rem;cursor:pointer">
        <option value="pending" ${o.status==='pending'?'selected':''}>รอดำเนินการ</option>
        <option value="doing" ${o.status==='doing'?'selected':''}>กำลังทำ</option>
        <option value="done" ${o.status==='done'?'selected':''}>ส่งแล้ว</option>
      </select></td>
    </tr>`).join('');
}

function changeStatus(id, status) {
  const o = orders.find(o => o.id===id);
  if (o) { o.status = status; refreshAdmin(); showAlert('✅ อัปเดตสถานะแล้ว'); }
}

// ── ALERT ──
let alertTimer;
function showAlert(msg) {
  const ex = document.querySelector('.alert');
  if (ex) ex.remove();
  clearTimeout(alertTimer);
  const el = document.createElement('div');
  el.className = 'alert';
  el.innerHTML = `<span>${msg}</span>`;
  document.body.appendChild(el);
  alertTimer = setTimeout(() => el.remove(), 3200);
}
